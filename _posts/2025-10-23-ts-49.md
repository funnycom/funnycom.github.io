export interface User {
  id: number;
  name: string;
  email: string;
}

export class UserManager {
  private users: User[] = [];
  
  addUser(user: User) {
    this.users.push(user);
  }
  
  getUser(id: number): User | undefined {
    return this.users.find(u => u.id === id);
  }
}
