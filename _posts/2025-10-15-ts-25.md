```typescript
/**
 * API 통신을 담당하는 서비스 클래스
 */
export class ApiService {
  private baseUrl: string;

  constructor(baseUrl: string = "https://jsonplaceholder.typicode.com") {
    this.baseUrl = baseUrl;
  }

  /**
   * GET 요청
   */
  async get<T>(endpoint: string): Promise<T> {
    try {
      const response = await fetch(`${this.baseUrl}${endpoint}`);
      
      if (!response.ok) {
        throw new Error(`HTTP error! status: ${response.status}`);
      }
      
      return await response.json();
    } catch (error) {
      console.error("GET request failed:", error);
      throw error;
    }
  }

  /**
   * POST 요청
   */
  async post<T>(endpoint: string, data: any): Promise<T> {
    try {
      const response = await fetch(`${this.baseUrl}${endpoint}`, {
        method: "POST",
        headers: {
          "Content-Type": "application/json"
        },
        body: JSON.stringify(data)
      });
      
      if (!response.ok) {
        throw new Error(`HTTP error! status: ${response.status}`);
      }
      
      return await response.json();
    } catch (error) {
      console.error("POST request failed:", error);
      throw error;
    }
  }

  /**
   * DELETE 요청
   */
  async delete(endpoint: string): Promise<void> {
    try {
      const response = await fetch(`${this.baseUrl}${endpoint}`, {
        method: "DELETE"
      });
      
      if (!response.ok) {
        throw new Error(`HTTP error! status: ${response.status}`);
      }
    } catch (error) {
      console.error("DELETE request failed:", error);
      throw error;
    }
  }
}
```

**managers/todo.manager.ts (기본 구조)**
```typescript
import { Todo, FilterType, SortType, Statistics } from "../types/todo.types";
import { ApiService } from "../services/api.service";

/**
 * Todo를 관리하는 매니저 클래스
 */
export class TodoManager {
  private todos: Todo[] = [];
  private nextId: number = 1;
  private apiService: ApiService;

  constructor(apiService: ApiService) {
    this.apiService = apiService;
  }

  /**
   * 새로운 할 일 추가
   */
  addTodo(title: string, userId: number = 1): Todo {
    if (!title.trim()) {
      throw new Error("제목은 비어있을 수 없습니다.");
    }

    const newTodo: Todo = {
      id: this.nextId++,
      title: title.trim(),
      completed: false,
      userId
    };

    this.todos.push(newTodo);
    return newTodo;
  }

  /**
   * 할 일 삭제
   */
  deleteTodo(id: number): boolean {
    const index = this.todos.findIndex(todo => todo.id === id);
    
    if (index === -1) {
      return false;
    }

    this.todos.splice(index, 1);
    return true;
  }

  /**
   * 완료 상태 토글
   */
  toggleTodo(id: number): boolean {
    const todo = this.todos.find(todo => todo.id === id);
    
    if (!todo) {
      return false;
    }

    todo.completed = !todo.completed;
    return true;
  }

  /**
   * 할 일 목록 조회 (필터링)
   */
  getTodos(filter: FilterType = "all"): Todo[] {
    switch (filter) {
      case "completed":
        return this.todos.filter(todo => todo.completed);
      case "active":
        return this.todos.filter(todo => !todo.completed);
      default:
        return [...this.todos];
    }
  }

  /**
   * 서버에서 할 일 가져오기
   */
  async fetchFromServer(userId?: number): Promise<number> {
    try {
      const endpoint = userId ? `/todos?userId=${userId}` : "/todos";
      const todos = await this.apiService.get<Todo[]>(endpoint);
      
      this.todos = todos;
      
      if (todos.length > 0) {
        this.nextId = Math.max(...todos.map(t => t.id)) + 1;
      }
      
      return todos.length;
    } catch (error) {
      console.error("Failed to fetch todos:", error);
      return 0;
    }
  }

  /**
   * 통계 정보 조회
   */
  getStatistics(): Statistics {
    const total = this.todos.length;
    const completed = this.todos.filter(t => t.completed).length;
    const active = total - completed;
    const completionRate = total > 0 ? (completed / total) * 100 : 0;

    return {
      total,
      completed,
      active,
      completionRate: Math.round(completionRate * 10) / 10
    };
  }
}
```

**index.ts (사용 예시)**
```typescript
import { ApiService } from "./services/api.service";
import { TodoManager } from "./managers/todo.manager";

async function main() {
  console.log("=== Todo 앱 시작 ===\n");

  // 1. ApiService와 TodoManager 생성
  const apiService = new ApiService();
  const todoManager = new TodoManager(apiService);

  // 2. 서버에서 데이터 가져오기
  console.log("서버에서 데이터 가져오는 중...");
  const count = await todoManager.fetchFromServer(1);
  console.log(`${count}개의 할 일을 가져왔습니다.\n`);

  // 3. 가져온 할 일 출력 (처음 5개)
  console.log("=== 할 일 목록 (처음 5개) ===");
  const todos = todoManager.getTodos();
  todos.slice(0, 5).forEach(todo => {
    const status = todo.completed ? "✅" : "⬜";
    console.log(`${status} [${todo.id}] ${todo.title}`);
  });

  // 4. 새 할 일 추가
  console.log("\n=== 새 할 일 추가 ===");
  const newTodo = todoManager.addTodo("TypeScript 마스터하기");
  console.log(`추가됨: [${newTodo.id}] ${newTodo.title}`);

  // 5. 할 일 완료 처리
  console.log("\n=== 할 일 완료 처리 ===");
  const toggleSuccess = todoManager.toggleTodo(newTodo.id);
  if (toggleSuccess) {
    console.log(`[${newTodo.id}] 완료 처리됨`);
  }

  // 6. 통계 출력
  console.log("\n=== 통계 ===");
  const stats = todoManager.getStatistics();
  console.log(`전체: ${stats.total}개`);
  console.log(`완료: ${stats.completed}개`);
  console.log(`미완료: ${stats.active}개`);
  console.log(`완료율: ${stats.completionRate}%`);

  // 7. 완료된 할 일만 출력
  console.log("\n=== 완료된 할 일 ===");
  const completedTodos = todoManager.getTodos("completed");
  completedTodos.slice(0, 5).forEach(todo => {
    console.log(`✅ [${todo.id}] ${todo.title}`);
  });

  console.log("\n=== Todo 앱 종료 ===");
}

// 실행
main().catch(console.error);
```
